80/// @brief adv广播属性bit位
81/// @details
82/// 若使能了bit位OB_ADV_PROP_BIT4_LEGACY_ADV，则使用传统广播，广播数据长度最大为31字节，广播属性仅支持如下组合:\n
83/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT1_SCANNABLE | OB_ADV_PROP_BIT0_CONNECTALBE\n
84/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT2_DIRECTED_ADV | OB_ADV_PROP_BIT0_CONNECTALBE\n
85/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT3_HIGH_DUTY_CYCLE | OB_ADV_PROP_BIT0_CONNECTALBE\n
86/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT1_SCANNABLE\n
87/// 若未使能bit位OB_ADV_PROP_BIT4_LEGACY_ADV，则使用扩展广播，广播数据长度最大为251字节\n
88enum ob_adv_properties_bits {
89 OB_ADV_PROP_BIT0_CONNECTALBE = 1 << 0, ///< 可连接
90 OB_ADV_PROP_BIT1_SCANNABLE = 1 << 1, ///< 响应scan request
91 OB_ADV_PROP_BIT2_DIRECTED_ADV = 1 << 2, ///< 直接广播
92 OB_ADV_PROP_BIT3_HIGH_DUTY_CYCLE = 1 << 3, ///< 高速直接广播
93 OB_ADV_PROP_BIT4_LEGACY_ADV = 1 << 4, ///< 传统广播
94 OB_ADV_PROP_BIT5_OMIT_ADV_ADDRESS = 1 << 5, ///< 匿名广播
95 OB_ADV_PROP_BIT6_INCLUDE_TX_POWER = 1 << 6, ///< 包含Tx Power
96};

98/// 预定义的adv广播属性
99enum ob_adv_properties {
100 OB_ADV_PROP_LEGACY_IND = 0x13, ///< 传统广播
101 OB_ADV_PROP_LEGACY_DIRECT_IND_LOW = 0x15, ///< 低速直连传统广播，广播间隔由参数定义
102 OB_ADV_PROP_LEGACY_DIRECT_IND_HIGH = 0x1D, ///< 高速直连传统广播，广播间隔由controller控制（小于3.75ms）
103 OB_ADV_PROP_LEGACY_SCAN_IND = 0x12, ///< 可扫描不可连接传统广播
104 OB_ADV_PROP_LEGACY_NONCONN_IND = 0x10, ///< 不可连接不可扫描传统广播
105 OB_ADV_PROP_EXT_CONN_NONSCAN = 0x01, ///< 可连接不可扫描扩展广播
106 OB_ADV_PROP_EXT_NONCONN_SCAN = 0x02, ///< 不可连接可扫描扩展广播
107 OB_ADV_PROP_EXT_NONCONN_NONSCAN = 0x00, ///< 不可连接不可扫描扩展广播
108 OB_ADV_PROP_EXT_ANONYMOUS = 0x20, ///< 匿名扩展广播（忽略广播地址）
