80/// @brief advertising attribute bit
81/// @details
82/// If the bit OB_ADV_PROP_BIT4_LEGACY_ADV is enabled, the legacy advertising is used, the maximum advertising data length is 31 bytes, and the advertising attribute only supports the following combinations:\n
83/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT1_SCANNABLE | OB_ADV_PROP_BIT0_CONNECTALBE\n
84/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT2_DIRECTED_ADV | OB_ADV_PROP_BIT0_CONNECTALBE\n
85/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT3_HIGH_DUTY_CYCLE | OB_ADV_PROP_BIT0_CONNECTALBE\n
86/// OB_ADV_PROP_BIT4_LEGACY_ADV | OB_ADV_PROP_BIT1_SCANNABLE\n
87/// If bit OB_ADV_PROP_BIT4_LEGACY_ADV is not enabled, extended advertising is used, and the maximum advertising data length is 251 bytes\n
88enum ob_adv_properties_bits {
89 OB_ADV_PROP_BIT0_CONNECTALBE = 1 << 0, ///< connectable
90 OB_ADV_PROP_BIT1_SCANNABLE = 1 << 1, ///< respond to scan request
91 OB_ADV_PROP_BIT2_DIRECTED_ADV = 1 << 2, ///< direct advertising
92 OB_ADV_PROP_BIT3_HIGH_DUTY_CYCLE = 1 << 3, ///< high-speed direct advertising
93 OB_ADV_PROP_BIT4_LEGACY_ADV = 1 << 4, ///< legacy advertising
94 OB_ADV_PROP_BIT5_OMIT_ADV_ADDRESS = 1 << 5, ///< anonymous advertising
95 OB_ADV_PROP_BIT6_INCLUDE_TX_POWER = 1 << 6, ///< Tx Power included
96};
