229__STATIC_FORCEINLINE void drv_audio_set_adc_vol(uint8_t vol)
230{
231 audio_state_t state = drv_audio_work_state();
232 if(state < AUDIO_STATE_INITED) {
233 return;
234 }
235
236#if RTE_AUDIO_USE_INTERNAL
237 // Wait updated last time
238 while(OM_AUDIO->ADC_VOL_CTRL & AU_VOL_UPDATE_MASK);
239 // Disable mute and unmute flow, bypass
240 register_set(&OM_AUDIO->ADC_VOL_CTRL, MASK_3REG(AU_VOL_ADCUNMU, 0, AU_VOL_ADCMU, 0, AU_VOL_MUTE_BYPASS, 1));
241 // Set new volume
242 register_set(&OM_AUDIO->ADC_VOL_CTRL, MASK_2REG(AU_VOL_LADC, vol, AU_VOL_UPDATE, 1));
243#else
244 (void)vol;
245#endif /* RTE_AUDIO_USE_INTERNAL */
246}

318__STATIC_FORCEINLINE void drv_audio_alc_config(bool alc_en, bool ng_en, audio_alc_config_t *config)
319{
320 audio_state_t state = drv_audio_work_state();
321 if(state < AUDIO_STATE_INITED) {
322 return;
323 }
324
325#if RTE_AUDIO_USE_INTERNAL
326 if(config) {
327 register_set(&OM_AUDIO->ALC_CTRL_1, MASK_4REG(AU_ALC_ATTACK, (config->at & 0x0F),
328 AU_ALC_DECAY, (config->dt & 0x0F),
329 AU_ALC_HOLD, (config->ht & 0x0F),
330 AU_ALC_NOISE_TYPE, (config->ntype & 0x01)));
331
332 register_set(&OM_AUDIO->ALC_CTRL_2, MASK_4REG(AU_ALC_TARGET, config->tar_gain,
333 AU_ALC_TARGET_JITTER, config->target_jitter,
334 AU_ALC_NGTH, config->ngth_gain,
335 AU_ALC_NOISE_JITTER, config->noise_jitter));
336 }
337
338 register_set(&OM_AUDIO->ALC_CTRL_1, MASK_2REG(AU_ALC_DIG_EN, alc_en, AU_ALC_NOISE_EN, ng_en));
339
340#else
341#endif /* RTE_AUDIO_USE_INTERNAL */
342}
