188/**
189 *******************************************************************************
190 * @brief FLASH write with interrupt enabled, the write interrupt step is:
191 * 1. drv_flash_write_int_start
192 * 2. drv_flash_write_int_status_get
193 * 3. if is_wip is 1, go to step 2, else go to step 4
194 * 4. drv_flash_write_int_continue
195 * 5. go to step 2
196 * 6. loop step 2 - 5,
197 * until the return value of step 4 is OM_ERROR_STATUS
198 *
199 * @param om_flash The FLASH controller device address
200 * @param addr The address of FLASH
201 * @param data The writing data buffer pointer in RAM
202 * Attention: the type must be volatile to avoid compiler optimization
203 * when Link Time Optimization is enabled,
204 * if the data is from heap, it should be sure to be volatile,
205 * for example: volatile uint8_t *data = om_mem_malloc(X, X);
206 * @param data_len The writing data length
207 *
208 * @return Error code, see@ref om_error_t
209 *******************************************************************************
210 */
211extern om_error_t drv_flash_write_int_start(OM_FLASH_Type om_flash,
212 uint32_t addr,
213 volatile uint8_t *data,
214 uint32_t data_len);
