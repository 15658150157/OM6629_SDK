193/**
194 *******************************************************************************
195 * @brief Get TIM base from tim idx
196 *
197 * @param idx Index of TIM peripheral
198 *
199 * @return OM_TIM Type pointer
200 *******************************************************************************
201 */
202static inline OM_TIM_Type* drv_tim_idx2base(uint8_t idx)
203{
204 OM_TIM_Type *const tim[] = {
205 #if (RTE_TIM0)
206 OM_TIM0,
207 #else
208 NULL,
209 #endif
210 #if (RTE_TIM1)
211 OM_TIM1,
212 #else
213 NULL,
214 #endif
215 #if (RTE_TIM2)
216 OM_TIM2,
217 #else
218 NULL,
219 #endif
220 };
221
222 return (idx < sizeof(tim)/sizeof(tim[0])) ? tim[idx] : NULL;
223}
