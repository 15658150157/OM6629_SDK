43#define GPIO_MASK(gpio_idx) (1U << (gpio_idx))

46/*******************************************************************************
47 * TYPEDEFS
48 */
49/// GPIO Mask: bit field for gpio, 1 indicates valid, 0 indicates invalid
50typedef uint32_t gpio_mask_t;
51
52/// GPIO Level
53typedef enum {
54 /// gpio level is low
55 GPIO_LEVEL_LOW = 0,
56 /// gpio level is high
57 GPIO_LEVEL_HIGH = 1,
58} gpio_level_t;
59
60/// GPIO Direction
61typedef enum {
62 /// GPIO output direction
63 GPIO_DIR_OUTPUT = 0U,
64 /// GPIO input direction
65 GPIO_DIR_INPUT = 1U,
66} gpio_dir_t;
67
68/// GPIO Interrupt Trigger Type
69typedef enum {
70 /// GPIO trigger type is falling edge
71 GPIO_TRIG_FALLING_EDGE = 0x0CU,
72 /// GPIO trigger type is rising edge
73 GPIO_TRIG_RISING_EDGE = 0x03U,
74 /// GPIO trigger type is both edge
75 GPIO_TRIG_RISING_FAILING_EDGE = 0x06U,
76 /// GPIO trigger type is low level
77 GPIO_TRIG_LOW_LEVEL = 0x00U,
78 /// GPIO trigger type is high level
79 GPIO_TRIG_HIGH_LEVEL = 0x0FU,
80 /// GPIO trigger type is none
81 GPIO_TRIG_NONE = 0x10U,
82} gpio_trig_type_t;
83
84/// GPIO Configuration
85typedef struct {
86 /// Pointer to OM_GPIOx
87 OM_GPIO_Type *om_gpio;
88 /// GPIO index in a GPIO port, range in [0, 38]
89 uint8_t gpio_idx;
90 /// GPIO direction
91 gpio_dir_t dir;
92 /// Used when dir config as output, range in [0, 1]
93 gpio_level_t out_val;
94 /// Used when dir config as input
95 gpio_trig_type_t trig_type;
96} gpio_config_t;
97
98/// GPIO Control
99typedef enum {
100 GPIO_CONTROL_CLEAR_INT = 0U, /*!< Clear GPIO interrupt, argu is gpio mask, return OM_ERROR_OK */
101} gpio_control_t;

107/**
108 *******************************************************************************
109 * @brief Set GPIO direction
110 *
111 * @param[in] om_gpio Pointer to GPIO
112 * @param[in] gpio_mask GPIO pin mask
113 * @param[in] dir GPIO direction
114 *******************************************************************************
115 */
116__STATIC_FORCEINLINE void drv_gpio_set_dir(OM_GPIO_Type *om_gpio, gpio_mask_t gpio_mask, gpio_dir_t dir)
117{
118 if (dir == GPIO_DIR_OUTPUT) {
119 om_gpio->OUTENSET = gpio_mask;
120 } else {
121 om_gpio->OUTENCLR = gpio_mask;
122 }
123}
124
125/**
126 *******************************************************************************
127 * @brief Get GPIO direction
128 *
129 * @param[in] om_gpio Pointer to GPIO
130 * @param[in] gpio_mask GPIO pin mask
131 *
132 * @return GPIO direction
133 *******************************************************************************
134 */
135__STATIC_FORCEINLINE gpio_dir_t drv_gpio_get_dir(OM_GPIO_Type *om_gpio, gpio_mask_t gpio_mask)
136{
137 return (om_gpio->OUTENSET & gpio_mask) ? GPIO_DIR_OUTPUT : GPIO_DIR_INPUT;
138}
139
140/**
141 *******************************************************************************
142 * @brief Read GPIO pin input
143 *
144 * @param[in] om_gpio Pointer to GPIO
145 * @param[in] gpio_mask GPIO pin mask
146 *
147 * @return GPIO pad value
148 *******************************************************************************
149 */
150__STATIC_FORCEINLINE uint32_t drv_gpio_read(OM_GPIO_Type *om_gpio, gpio_mask_t gpio_mask)
151{
152 return (om_gpio->DATA & gpio_mask);
153}
154
155/**
156 *******************************************************************************
157 * @brief Write GPIO pin output
158 *
159 * @param[in] om_gpio Pointer to GPIO
160 * @param[in] gpio_mask GPIO pin mask
161 * @param[in] val GPIO pin value
162 *******************************************************************************
163 */
164__STATIC_FORCEINLINE void drv_gpio_write(OM_GPIO_Type *om_gpio, gpio_mask_t gpio_mask, gpio_level_t val)
165{
166 OM_CRITICAL_BEGIN();
167 if (val) {
168 om_gpio->DATAOUT |= gpio_mask;
169 } else {
170 om_gpio->DATAOUT &= ~gpio_mask;
171 }
172 OM_CRITICAL_END();
173}
174
175/**
176 *******************************************************************************
177 * @brief Toggle GPIO pin output
178 *
179 * @param[in] om_gpio Pointer to GPIO
180 * @param[in] gpio_mask GPIO pin mask
181 *******************************************************************************
182 */
183__STATIC_FORCEINLINE void drv_gpio_toggle(OM_GPIO_Type *om_gpio, gpio_mask_t gpio_mask)
184{
185 OM_CRITICAL_BEGIN();
186 om_gpio->DATAOUT ^= gpio_mask;
187 OM_CRITICAL_END();
188}

280/**
281 *******************************************************************************
282 * @brief Get GPIO base from gpio idx
283 *
284 * @param idx Index of GPIO peripheral
285 *
286 * @return OM_GPIO Type pointer
287 *******************************************************************************
288 */
289static inline OM_GPIO_Type* drv_gpio_idx2base(uint8_t idx)
290{
291 OM_GPIO_Type *const gpio[] = {
292 #if (RTE_GPIO0)
293 OM_GPIO0,
294 #else
295 NULL,
296 #endif
297 #if (RTE_GPIO1)
298 OM_GPIO1,
299 #else
300 NULL,
301 #endif
302 };
303
304 return (idx < sizeof(gpio)/sizeof(gpio[0])) ? gpio[idx] : NULL;
305}

307/**
308 *******************************************************************************
309 * @brief Control GPIO interface.
310 *
311 * @param[in] om_gpio Pointer to GPIO port, Donot GPIO_CONTROL_CLK_DISABLE if using sleep
312 * @param[in] control Operation
313 * @param[in] argu Not used, always set NULL
314 *
315 * @return Control status, always return OM_ERROR_OK
316 *******************************************************************************
317 */
318static inline void *drv_gpio_control(OM_GPIO_Type *om_gpio, gpio_control_t control, void *argu)
319{
320 switch (control) {
321 case GPIO_CONTROL_CLEAR_INT:
322 om_gpio->INTSTATUS = (uint32_t)argu;
323 break;
324 default:
325 break;
326 }
327
328 return (void *)OM_ERROR_OK;
329}
