43/// LP Timer Control
44typedef enum {
45 LPTIM_CONTROL_ENABLE, /*!< When argu is NULL, disable lptim, and vice versa */
46 LPTIM_CONTROL_START, /*!< Start LP Timer */
47 LPTIM_CONTROL_STOP, /*!< Stop LP Timer */
48 LPTIM_CONTROL_CLEAR, /*!< Clear LP Timer counter, and load top val */
49 LPTIM_CONTROL_CTO0, /*!< Clear toggle out0 to its idle state */
50 LPTIM_CONTROL_CTO1, /*!< Clear toggle out1 to its idle state */
51 LPTIM_CONTROL_CTO2, /*!< Clear toggle out2 to its idle state */
52 LPTIM_CONTROL_CTO3, /*!< Clear toggle out3 to its idle state */
53 LPTIM_CONTROL_CNT_SET, /*!< Set counter val, val is in range [0~0xFFFF] */
54 LPTIM_CONTROL_TOP_SET, /*!< Set top val, val is in range [0~0xFFFF] */
55 LPTIM_CONTROL_REP0_SET, /*!< Set rep0 val, val is in range [0~0xFF] */
56 LPTIM_CONTROL_REP1_SET, /*!< Set rep1 val, val is in range [0~0xFF] */
57 LPTIM_CONTROL_REP2_SET, /*!< Set rep0 val, val is in range [0~0xFF] */
58 LPTIM_CONTROL_REP3_SET, /*!< Set rep1 val, val is in range [0~0xFF] */
59 LPTIM_CONTROL_POWER_IN_SLEEP, /*!< When argu is not NULL, power on lptim in sleep, and vice versa */
60 LPTIM_CONTROL_INT_EN, /*!< Control enable or disable lptim interrupt, argu is the interrupt en combinations */
61} lptim_control_t;

63/// LP Timer Stop Modeï¼šRepeat Mode for two group output channel 23 and 01
64typedef enum {
65 /// Free Mode: When started, the LETIMER counts down until it is stopped by software
66 LPTIM_STOP_MODE_FREE = 0U,
67 /// stop01 Mode: The counter counts only while group 01 are running. When group01 stop running, the counter stops
68 LPTIM_STOP_MODE_STOP_01 = 1U,
69 /// stopor Mode: When group01 or group23 stop running, the counter stops.
70 LPTIM_STOP_MODE_STOP_OR = 2U,
71 /// stopboth Mode: When group01 and group23 both stop running, the counter stops.
72 LPTIM_STOP_MODE_STOPBOTH = 3U,
73} lptim_stop_mode_t;

310 *******************************************************************************
311 * @brief Get LPTIM base from lptim idx
312 *
313 * @param idx Index of LPTIM peripheral
314 *
315 * @return OM_LPTIM Type pointer
316 *******************************************************************************
317 */
318static inline OM_LPTIM_Type* drv_lptim_idx2base(uint8_t idx)
319{
320 OM_LPTIM_Type *const lptim[] = {
321 #if (RTE_LPTIM)
322 OM_LPTIM,
323 #else
324 NULL,
325 #endif
326 };
327
328 return (idx < sizeof(lptim)/sizeof(lptim[0])) ? lptim[idx] : NULL;
329}
