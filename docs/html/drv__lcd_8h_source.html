44/**
45 \brief Typedef of LCD control enumerate
46 */
47typedef enum {
48 LCD_CONTROL_SET_DATA_WIDTH = 0U, /**< With 1 argument range in 8/16/32 8: common command and data,
49 16: for rgb565
50 32: for rgb666 or rgb888 */
51 LCD_CONTROL_TX_DATA_MODE,
52 LCD_CONTROL_SET_CMD_BIT, /**< with 1 argument, range in 1/2/4, x lines to send command */
53 LCD_CONTROL_SET_DATA_BIT, /**< with 1 argument, range in 1/2/4, x lines to data bits */
54 LCD_CONTROL_SET_BUF_MODE_666_888, /**< No argument, for rgb666 or rgb888, trule value is 3 */
55 LCD_CONTROL_SET_BUF_MODE_COMMON, /**< No argument, value is 4 */
56 LCD_CONTROL_KEEP_CS,
57 LCD_CONTROL_RELEASE_CS,
58 LCD_CONTROL_SET_DUMMY_CYCLES, /**< With 1 argument, dummy cycles. range in[0, 0x7F] */
59 LCD_CONTROL_SET_CS_CONFIG, /**< With 1 argument, pointer to lcd_cs_config_t */
60} lcd_control_t;
61
62/**
63 \brief Typedef of LCD rgb mode enumerate
64 */
65typedef enum {
66 LCD_RGB_MODE_565 = 1U,
67 LCD_RGB_MODE_666 = 2U,
68 LCD_RGB_MODE_888 = 3U,
69} lcd_rgb_mode_t;
70
71/**
72 \brief Typedef of LCD data mode enumerate
73 */
74typedef enum {
75 LCD_FLASH_LIKE = 0U, /// target lcd like flash
76 LCD_3WM1 = 1U, /// target lcd using one more bit in cmd/data to differentiate whether it is cmd or data
77 LCD_3WM2 = 2U, /// target lcd 3WM1 can enable data_2_lan to send data in 2 lines, 3WM2 can not.
78 LCD_4WM1 = 3U, /// target lcd using io2 to differentiate whether it is cmd or data
79 LCD_4WM2 = 4U, /// target lcd 4WM1 can enable data_2_lan to send data in 2 lines, 4WM2 can not.
80} lcd_data_mode_t;
81
82/**
83 \brief Typedef of LCD spi mode enumerate
84 */
85typedef enum {
86 LCD_SPI_MODE_0 = 0U, // Mode 0: CPOL=0, CPHA=0
87 LCD_SPI_MODE_1 = 1U, // Mode 1: CPOL=0, CPHA=1
88 LCD_SPI_MODE_2 = 2U, // Mode 2: CPOL=1, CPHA=0
89 LCD_SPI_MODE_3 = 3U, // Mode 3: CPOL=1, CPHA=1
90} lcd_spi_mode_t;
91
