171/**
172 *******************************************************************************
173 * @brief calculate counter increment by pmu timer frequency
174 *
175 * @param[in] real_freq real frequency
176 *
177 * @return incr
178 *******************************************************************************
179 */
180__STATIC_FORCEINLINE uint32_t pmu_timer_calc_incr_by_real_freq(uint32_t real_freq)
181{
182 uint32_t old_incr, new_incr;
183 uint64_t tmp_val;
184 uint32_t tmp_val_lo, tmp_val_hi;
185
186 old_incr = OM_PMU->TIMER_PPM & 0x1FFFF;
187
188 tmp_val = (uint64_t)old_incr * (((uint64_t)32768 << 16) / real_freq);
189
190 // binary point position is at [bit32 . bit31]
191 tmp_val_lo = tmp_val & 0xFFFFFFFFU;
192 tmp_val_hi = (tmp_val >> 32) & 0xFFFFFFFFU;
193
194 new_incr = ((tmp_val_hi & 0x1) << 16) | (tmp_val_lo >> 16);
195
196 OM_ASSERT(new_incr != 0U);
197
198 return new_incr;
199}

201/**
202 *******************************************************************************
203 * @brief calculate counter increment by recording pmu timer's count value within time_us
204 *
205 * @param[in] time_us record time
206 * @param[in] pmu_tim_cnt record count
207 *
208 * @return incr
209 *******************************************************************************
210 */
211__STATIC_FORCEINLINE uint32_t pmu_timer_calc_incr_by_time(uint32_t time_us, uint32_t pmu_tim_cnt)
212{
213 return pmu_timer_calc_incr_by_real_freq((uint64_t)(pmu_tim_cnt) * 1000000 / time_us);
214}

228/**
229 *******************************************************************************
230 * @brief pmu timer control
231 *
232 * @param[in] trig_type trigger type
233 * @param[in] control control type
234 * @param[in] argu control content
235 *
236 * @return control status
237 *******************************************************************************
238 */
239__STATIC_FORCEINLINE void *drv_pmu_timer_control(pmu_timer_trig_t trig_type, pmu_timer_control_t control, void *argu)
240{
241 OM_PMU_Type *pmu_timer = OM_PMU;
242
243 switch (control) {
244 case PMU_TIMER_CONTROL_DISABLE:
245 OM_CRITICAL_BEGIN();
246 if (trig_type == PMU_TIMER_TRIG_VAL0) {
247 pmu_timer->TIMER_CTRL &= ~PMU_TIMER_CTRL_PMU_TIMER_INT_VAL0_EN_MASK;
248 } else if (trig_type == PMU_TIMER_TRIG_VAL1) {
249 pmu_timer->TIMER_CTRL &= ~PMU_TIMER_CTRL_PMU_TIMER_INT_VAL1_EN_MASK;
250 }
251 #if (PMU_TIMER_DEBUG)
252 pmu_timer_env.record[trig_type].is_enabled = false;
253 #endif
254 OM_CRITICAL_END();
255 break;
256 case PMU_TIMER_CONTROL_GET_TIMER_VAL:
257 if (trig_type == PMU_TIMER_TRIG_VAL0) {
258 return (void *)pmu_timer_env.TIMER_SET0;
259 } else if (trig_type == PMU_TIMER_TRIG_VAL1) {
260 return (void *)pmu_timer_env.TIMER_SET1;
261 }
262 break;
263 case PMU_TIMER_CONTROL_SET_TIMER_VAL:
264 OM_CRITICAL_BEGIN();
265 if (trig_type == PMU_TIMER_TRIG_VAL0) {
266 pmu_timer->TIMER_SET0 = pmu_timer_env.TIMER_SET0 = (uint32_t)argu;
267 REGW1(&pmu_timer->TIMER_CTRL, PMU_TIMER_CTRL_PMU_TIMER_INT_VAL0_EN_MASK);
268 } else if (trig_type == PMU_TIMER_TRIG_VAL1) {
269 pmu_timer->TIMER_SET1 = pmu_timer_env.TIMER_SET1 = (uint32_t)argu;
270 REGW1(&pmu_timer->TIMER_CTRL, PMU_TIMER_CTRL_PMU_TIMER_INT_VAL1_EN_MASK);
271 }
272#ifdef PMU_TIMER_DEBUG
273 pmu_timer_env.record[trig_type].is_enabled = true;
274 pmu_timer_env.record[trig_type].cur_cnt_when_set = drv_pmu_timer_cnt_get();
275 pmu_timer_env.record[trig_type].cur_cnt_when_overflow = 0;
276 pmu_timer_env.record[trig_type].value = trig_type==PMU_TIMER_TRIG_VAL0 ? pmu_timer_env.TIMER_SET0 : pmu_timer_env.TIMER_SET1;
277#endif
278 OM_CRITICAL_END();
279 break;
280 case PMU_TIMER_CONTROL_GET_OVERFLOW:
281 if (trig_type == PMU_TIMER_TRIG_VAL0) {
282 return (void *)(pmu_timer->TIMER_CTRL & PMU_TIMER_CTRL_PMU_TIMER_INT0_MASK);
283 } else if (trig_type == PMU_TIMER_TRIG_VAL1) {
284 return (void *)(pmu_timer->TIMER_CTRL & PMU_TIMER_CTRL_PMU_TIMER_INT1_MASK);
285 }
286 break;
287 case PMU_TIMER_CONTROL_SET_TIMER_INCR:
288 pmu_timer->TIMER_PPM = (uint32_t)argu;
289 break;
290 default:
291 break;
292 }
293
294 return (void *)((uint32_t)OM_ERROR_OK);
295}
