141 QDEC_CONTROL_SET_INIT_A = 8U, /*!< Set phase A first sample value after power on, return OM_ERROR_OK */
142 QDEC_CONTROL_SET_INIT_B = 9U, /*!< Set phase B first sample value after power on, return OM_ERROR_OK */
143 QDEC_CONTROL_GET_INIT_A = 10U, /*!< Get phase A first sample value after power on, return sample value A */
144 QDEC_CONTROL_GET_INIT_B = 11U, /*!< Get phase B first sample value after power on, return sample value B */
145 QDEC_CONTROL_GET_INPUT_A = 12U, /*!< Get phase A last sample value before power down, return sample value A */
146 QDEC_CONTROL_GET_INPUT_B = 13U, /*!< Get phase B last sample value before power down, return sample value B */
147} qdec_control_t;

189/**
190 *******************************************************************************
191 * @brief Control QDEC interface.
192 *
193 * @param[in] om_qdec Pointer to QDEC
194 * @param[in] control Operation
195 * @param[in] argu Operation argument
196 *
197 * @return status: Control status
198 *******************************************************************************
199 */
200__STATIC_FORCEINLINE void *drv_qdec_control(OM_QDEC_Type *om_qdec, qdec_control_t control, void *argu)
201{
202 uint32_t ret;
203
204 ret = (uint32_t)OM_ERROR_OK;
205
206 OM_CRITICAL_BEGIN();
207 switch (control) {
208 case QDEC_CONTROL_START:
209 // Start the QDEC
210 REGW(&om_qdec->START, MASK_1REG(QDEC_START_START, 1U));
211 break;
212 case QDEC_CONTROL_STOP:
213 // Stop the QDEC
214 REGW(&om_qdec->STOP, MASK_1REG(QDEC_STOP_STOP, 1U));
215 break;
216 case QDEC_CONTROL_SET_SAMPLEPER:
217 REGW(&om_qdec->SAMPLEPER, MASK_1REG(QDEC_SAMPLEPER_SAMPLEPER, (uint32_t)argu));
218 break;
219 case QDEC_CONTROL_SET_REPORTPER:
220 REGW(&om_qdec->REPORTPER, MASK_1REG(QDEC_REPORTPER_REPORTPER, (uint32_t)argu));
221 break;
222 case QDEC_CONTROL_SET_DBFEN:
223 REGW(&om_qdec->DBFEN, MASK_1REG(QDEC_DBFEN_DBFEN, (uint32_t)argu));
224 break;
225 case QDEC_CONTROL_SET_LEDPOL:
226 REGW(&om_qdec->LEDPOL, MASK_1REG(QDEC_LEDPOL_LEDPOL, (uint32_t)argu));
227 break;
228 case QDEC_CONTROL_SET_LEDPRE:
229 REGW(&om_qdec->LEDPRE, MASK_1REG(QDEC_LEDPRE_LEDPRE, (uint32_t)argu));
230 break;
231 case QDEC_CONTROL_SET_PSELLED:
232 REGW(&om_qdec->PSELLED, MASK_1REG(QDEC_PSELLED_CONNECT, (uint32_t)argu));
233 break;
234 case QDEC_CONTROL_SET_INIT_A:
235 REGW(&om_qdec->INIT, MASK_1REG(QDEC_INIT_A, (uint32_t)argu));
236 break;
237 case QDEC_CONTROL_SET_INIT_B:
238 REGW(&om_qdec->INIT, MASK_1REG(QDEC_INIT_B, (uint32_t)argu));
239 break;
240 case QDEC_CONTROL_GET_INIT_A:
241 ret = REGR(&om_qdec->INIT, MASK_POS(QDEC_INIT_A));
242 break;
243 case QDEC_CONTROL_GET_INIT_B:
244 ret = REGR(&om_qdec->INIT, MASK_POS(QDEC_INIT_B));
245 break;
246 case QDEC_CONTROL_GET_INPUT_A:
247 ret = REGR(&om_qdec->INPUT, MASK_POS(QDEC_INPUT_A));
248 break;
249 case QDEC_CONTROL_GET_INPUT_B:
250 ret = REGR(&om_qdec->INPUT, MASK_POS(QDEC_INPUT_B));
251 break;
252 default:
253 break;
254 }
255
256 OM_CRITICAL_END();
257
258 return (void *)ret;
259}
