124/// OM24G config
125typedef struct {
126 /// tx data pointer
127 uint8_t *tx_data;
128 /// rx data pointer
129 uint8_t *rx_data;
130 // select package structure A or B
131 om24g_structure_t packet_struct_sel;
132 /// preamble content
133 uint8_t preamble;
134 /// preamble lenth
135 uint8_t preamble_len;
136 /// select sync word0 or word1 to transmit package
137 om24g_syncword_t sync_word_sel;
138 /// sync word0
139 uint32_t sync_word0;
140 /// sync word1
141 uint32_t sync_word1;
142 /// tx pipe address
143 uint8_t tx_addr;
144 /// rx pipe0 address
145 uint8_t rx_addr;
146 /// If it is 1, check if rx pipe address matches, and if it is 0, do not check.
147 uint8_t addr_chk;
148 /// The number of bits in the header.
149 uint8_t hdr_bits;
150 /// In fixed length mode, it is the length of the receiving and sending. In dynamic length mode, it is the maximum packet length that can be sent.
151 uint16_t static_len;
152 /// The number of bits in rx pipe address.
153 uint8_t addr1_bits;
154 /// The number of bits in lenth.
155 uint8_t len_bits;
156 /// The location of rx pipe address.
157 uint8_t addr1_pos;
158 /// The location of rx pipe address(after/in/before header).
159 uint8_t addr1_loc;
160 /// The location of lenth.
161 uint8_t len_pos;
162 /// MSB first or LSB first transmit.
163 om24g_endian_t endian;
164 /// The channel for transmitting frequency points.
165 uint16_t freq;
166 /// air date rate.
167 om24g_rate_t data_rate;
168 /// enable ack mode.
169 uint8_t ack_en;
170 /// Enable dynamic length.
171 uint8_t dpl_en;
172 /// Enable whitening.
173 uint8_t white_en;
174 /// 1: Do not whiten the header.
175 uint8_t white_skip_hdr;
176 /// 1: Do not whiten the pipe address.
177 uint8_t white_skip_addr;
178 /// 1: Do not whiten CRC.
179 uint8_t white_skip_crc;
180 /// select whiten mode(0~7) to Adapt to different chips.
181 uint8_t white_sel;
182 /// white seed value.
183 uint16_t white_seed;
184 /// white out bit.
185 uint8_t white_obit;
186 /// lenth of crc.
187 om24g_crc_byte_t crc_len;
188 /// Enable crc.
189 uint8_t crc_en;
190 /// select crc mode.
191 uint8_t crc_mode;
192 /// CRC Polynomial
193 uint32_t crc_poly;
194 /// CRC initital value.
195 uint32_t crc_init;
196 /// 1: CRC does not verify synchronization words.
197 uint8_t crc_skip_sync;
198 /// 1: CRC does not verify lenth.
199 uint8_t crc_skip_len;
200 /// 1: CRC does not verify pipe address.
201 uint8_t crc_skip_addr;
202 /// Select GFSK or FSK.
203 om24g_modulation_t modulation_mode;
204 /// Select modulation mode.
205 om24g_detect_mode_t detect_mode;
206} om24g_config_t;
