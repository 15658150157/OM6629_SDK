43/// I2C Mode
44typedef enum {
45 /// Master Mode, 7-bit addressing mode
46 I2C_MODE_MASTER = 0,
47 /// Slave Mode, 7-bit addressing mode
48 I2C_MODE_SLAVE = 1,
49 /// Slave Mode, 10-bit addressing mode
50 I2C_MODE_SMBUS_DEVICE = 2,
51 /// Master Mode, 10-bit addressing mode
52 I2C_MODE_SMBUS_HOST = 3,
53} i2c_mode_t;
54
55/// I2C Speed
56typedef enum {
57 /// Speed: 100khz
58 I2C_SPEED_100K = 1,
59 /// Speed: 400khz
60 I2C_SPEED_400K = 2,
61 /// Speed: 1Mhz
62 I2C_SPEED_1M = 3,
63 /// Speed: 2Mhz
64 I2C_SPEED_2M = 4,
65 /// Speed: MAX
66 I2C_SPEED_MAX = 5,
67} i2c_speed_t;
68
69/// I2C Config
70typedef struct {
71 i2c_mode_t mode;
72 i2c_speed_t speed;
73} i2c_config_t;
74
75/// I2C Control
76typedef enum {
77 /// check I2C is busy
78 I2C_CONTROL_IS_BUSY = 0U,
79 /// check device is valid, argu is i2c_addr

81} i2c_control_t;

87/**
88 *******************************************************************************
89 * @brief Get I2C base from i2c idx
90 *
91 * @param[in] idx Index of I2C peripheral
92 *
93 * @return OM_I2C Type pointer
94 *******************************************************************************
95 */
96static inline OM_I2C_Type* drv_i2c_idx2base(uint8_t idx)
97{
98 OM_I2C_Type *const i2c[] = {
99 #if (RTE_I2C0)
100 OM_I2C0,
101 #else
102 NULL,
103 #endif
104 #if (RTE_I2C1)
105 OM_I2C1,
106 #else
107 NULL,
108 #endif
109 };
110
111 return (idx < sizeof(i2c)/sizeof(i2c[0])) ? i2c[idx] : NULL;
112}

393 *******************************************************************************
394 * @brief Get I2C TX ABRT SOURCE
395 *
396 * @param[in] om_i2c Pointer to I2C
397 *
398 * @return I2C TX ABRT SOURCE value
399 *******************************************************************************
400 */
401__STATIC_FORCEINLINE uint32_t drv_i2c_get_tx_abrt(OM_I2C_Type *om_i2c)
402{
403 uint32_t result;
404 uint32_t state;
405 result = (om_i2c->RAW_INTR_STAT & I2C_INTR_TX_ABRT_MASK)&&(om_i2c->TX_ABRT_SOURCE & (I2C_TX_ABRT_SRC_7B_ADDR_NOACK_MASK | I2C_TX_ABRT_SRC_10ADDR1_NOACK_MASK | I2C_TX_ABRT_SRC_10ADDR2_NOACK_MASK | I2C_TX_ABRT_SRC_TXDATA_NOACK_MASK));
406 state = (result != 0) ? 1 : 0;
407 return (state);
408}
