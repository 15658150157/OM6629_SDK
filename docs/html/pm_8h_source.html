54/// PM checker priority

58 PM_CHECKER_PRIORITY_NORMAL = 2, ///< Normal
59 PM_CHECKER_PRIORITY_HIGH = 3, ///< High
60 PM_CHECKER_PRIORITY_HIGHEST = 4, ///< Highest
61} pm_checker_priority_t;
62
63/// PM status
64typedef enum {
65 /// All modules are alive, and CPU run with full speed
66 PM_STATUS_ACTIVE,
67 /// All modules are alive, but CPU clock is gating
68 PM_STATUS_IDLE,
69 /// Power down most of module(CPU, Peripheral, etc),
70 /// but 32K is alive, only gpio and sleep-timer can wake up chip
71 PM_STATUS_SLEEP,
72 /// Power down most of module(CPU, Peripheral, etc),
73 /// 32K is not alive, only gpio can wake up chip
74 PM_STATUS_DEEP_SLEEP,
75} pm_status_t;
76
77/// Sleep state
78typedef enum {
79 /// Event with sleep entering.
80 /// The output of IO can be controlled.
81 PM_SLEEP_ENTRY,
82 /// Event with sleep leaving, CHIP is working on 'High Speed Internal Clock (RC32M)'.
83 /// The output of IO can not be controlled.
84 PM_SLEEP_LEAVE_TOP_HALF_HSI,
85 /// Event with sleep leaving, CHIP is working on 'High Speed External Clock (XTAL32M)'.
86 /// The output of IO can not be controlled.
87 PM_SLEEP_LEAVE_TOP_HALF,
88 /// Event with sleep leaving, CHIP is working on 'High Speed External Clock (XTAL32M)'.
89 /// The output of IO can be controlled.
90 PM_SLEEP_LEAVE_BOTTOM_HALF,
91
92 /// Alias for PM_SLEEP_ENTRY, just for driver REG store
93 PM_SLEEP_STORE = PM_SLEEP_ENTRY,
94 /// Alias for PM_SLEEP_LEAVE_TOP_HALF_HSI, just for driver REG restore
95 PM_SLEEP_RESTORE_HSI = PM_SLEEP_LEAVE_TOP_HALF_HSI,
96 /// Alias for PM_SLEEP_LEAVE_TOP_HALF, just for driver REG restore
97 PM_SLEEP_RESTORE_HSE = PM_SLEEP_LEAVE_TOP_HALF,
98} pm_sleep_state_t;
