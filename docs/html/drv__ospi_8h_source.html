45 * MACROS

47/**
48 * There are four phase in a frame
49 * 1. command phase
50 * 2. address phase
51 * 3. dummy cycle phase
52 * 4. data phase
53 * Except dummy cycle phase, each phase can be configured as
54 * different bus width, which means parallel data number, and
55 * bit count, which means data bit count number.
56 * Dummy cycle phase just indicate the clock latency between
57 * address phase and data phase.
58 */
59#define OSPI_FRAME_CONFIG(cmd_width, cmd_bits, addr_width, addr_bits, dummy, data_width) \
60 { \
61 ((cmd_width & 0xFF) << OSPI_SW_CFG0_P0_BUS_WIDTH_POS) | \
62 ((cmd_bits & 0xFF) << OSPI_SW_CFG0_P0_BIT_CNT_POS) | \
63 ((addr_width & 0xFF) << OSPI_SW_CFG0_P1_BUS_WIDTH_POS) | \
64 ((addr_bits & 0xFF) << OSPI_SW_CFG0_P1_BIT_CNT_POS) | \
65 ((dummy & 0xFF) << OSPI_SW_CFG0_DUMMY_CYCLE_POS), \
66 ((data_width & 0xFF) << OSPI_SW_CFG1_SDATA_BUS_WIDTH_POS) | \
67 ((4U & 0xFF) << OSPI_SW_CFG1_BUF_WIDTH_BYTES_POS) | \
68 ((1U & 0x1) << OSPI_SW_CFG1_FIXED_LIST_EN_POS) | \
69 ((0U & 0xFF) << OSPI_SW_CFG1_HB_POS) | \
70 ((1U & 0xFF) << OSPI_SW_CFG1_EN_POS) \
71 }

279/**
280 *******************************************************************************
281 * @brief Control OSPI interface.
282 *
283 * @param[in] om_ospi Pointer to OSPI
284 * @param[in] control Operation
285 * @param[in] argu Operation argument
286 *
287 * @return status: Control status, always return OM_ERROR_OK
288 *******************************************************************************
289 */
290__STATIC_INLINE void *drv_ospi_control(OM_OSPI_Type *om_ospi, ospi_control_t control, void *argu)
291{
292 switch (control) {
293 case OSPI_CONTROL_CRYPT_CTRL:
294 if ((uint32_t)argu & OSPI_SEC_CTRL_ENC_MASK) {
295 DRV_RCC_CLOCK_ENABLE(RCC_CLK_AES, 1U);
296 }
297 om_ospi->SEC_CTRL = (uint32_t)argu;
298 break;
299 case OSPI_CONTROL_CS_CONFIG:
300 OM_ASSERT(argu);
301 do {
302 ospi_cs_config_t *cs_config;
303 cs_config = (ospi_cs_config_t *)argu;
304 om_ospi->CFG[0].CS_CFG = ((cs_config)->cs_active_high << OSPI_CS_CFG_CSP_POS)
305 | ((cs_config)->transp_head_ignore_cnt << OSPI_CS_CFG_TRSP_IGNORE_CNT_POS)
306 | ((cs_config)->cs_setup << OSPI_CS_CFG_CS_SETUP_POS)
307 | ((cs_config)->cs_hold << OSPI_CS_CFG_CS_HOLD_POS)
308 | ((cs_config)->cs_recover << OSPI_CS_CFG_CS_RECOVER_POS);
309 } while(0);
310 break;
311 case OSPI_CONTROL_SET_CLK_PARAM:
312 OM_ASSERT(argu);
313 do {
314 uint32_t cfg;
315 uint8_t cpol, cpha;
316 ospi_clk_param_t clk_param;
317 uint8_t falling_sample, bypass_clk_div_en;
318
319 clk_param.param_all = (uint32_t)argu;
320 cfg = om_ospi->CFG[0].SPI_CFG;
321 cpol = (cfg & OSPI_CFG_CPOL_MASK) ? 1U : 0U;
322 cpha = (cfg & OSPI_CFG_CPHA_MASK) ? 1U : 0U;
323 falling_sample = ((clk_param.clk_div < 2U) || (cpol != cpha)) ? 0U : clk_param.falling_sample;
324 bypass_clk_div_en = (clk_param.clk_div < 2) ? 1U : 0U;
325 register_set(&om_ospi->CFG[0].SPI_CFG, MASK_4REG(OSPI_CFG_CLK_DIV, clk_param.clk_div,
326 OSPI_CFG_BPC, bypass_clk_div_en,
327 OSPI_CFG_DLY, clk_param.delay_sample,
328 OSPI_CFG_FDS, falling_sample));
329 } while(0);
330 break;
331 case OSPI_CONTROL_GET_CLK_PARAM:
332 OM_ASSERT(argu);
333 do {
334 ospi_clk_param_t *clk_param = (ospi_clk_param_t *)argu;
335 uint32_t cfg;
336 cfg = om_ospi->CFG[0].SPI_CFG;
337 clk_param->clk_div = (cfg & OSPI_CFG_CLK_DIV_MASK) >> OSPI_CFG_CLK_DIV_POS;
338 clk_param->delay_sample = (cfg & OSPI_CFG_DLY_MASK) >> OSPI_CFG_DLY_POS;
339 clk_param->falling_sample = (cfg & OSPI_CFG_FDS_MASK) >> OSPI_CFG_FDS_POS;
340 } while(0);
341 break;
342 case OSPI_CONTROL_SET_RD_OPCODE:
343 OM_ASSERT(argu);
344 do {
345 ospi_frame_t *frame = (ospi_frame_t *)argu;
346 register_set(&om_ospi->OPCODE, MASK_1REG(OSPI_OPCODE_RD0, frame->cmd));
347 om_ospi->SW_CFG0 = frame->frame_cfg[0];
348 om_ospi->SW_CFG1 = frame->frame_cfg[1];
349 } while(0);
350 break;
351 case OSPI_CONTROL_SET_WR_OPCODE:
352 OM_ASSERT(argu);
353 do {
354 ospi_frame_t *frame = (ospi_frame_t *)argu;
355 register_set(&om_ospi->OPCODE, MASK_1REG(OSPI_OPCODE_WR0, frame->cmd));
356 om_ospi->SW_WR_CFG0 = frame->frame_cfg[0];
357 om_ospi->SW_WR_CFG1 = frame->frame_cfg[1];
358 } while(0);
359 break;
360 default:
361 break;
362 }
363
364 return (void *)OM_ERROR_OK;
365}
