44/// Max trans bytes = max_trans_size * src_width
45#define GPDMA_CHAN_MAX_TRANS_SIZE (0x3FFFFF)
46/// DMA Control Setting
47#define GPDMA_SET_CTRL(src_addr_ctrl, dst_addr_ctrl, src_width, dst_width, src_burst_size, priority) \
48 (((src_addr_ctrl) << GPDMA_CHAN_CTRL_SRCADDRCTRL_POS) | \
49 ((dst_addr_ctrl) << GPDMA_CHAN_CTRL_DSTADDRCTRL_POS) | \
50 ((src_width) << GPDMA_CHAN_CTRL_SRCWIDTH_POS) | \
51 ((dst_width) << GPDMA_CHAN_CTRL_DSTWIDTH_POS) | \
52 ((src_burst_size) << GPDMA_CHAN_CTRL_SRCBURSTSIZE_POS) | \
53 ((priority) << GPDMA_CHAN_CTRL_PRIORITY_POS))
54
55#define GPDMA_INVALID_HANDSHAKE_SIGNAL 0xFFU /* invalid handshake signal */

57/**
58 *******************************************************************************
59 * @brief RAM address must be upper 0x20000000U if GPDMA used to transfer to/from
60 * RAM memory. This macro used to convert RAM address to upper 0x20000000U
61 * If address not in lowwer RAM address, the macro donnot convert. This macro
62 * can be used to GPDMA chain struct member.
63 *
64 * @param[in] addr GPDMA source or dst address.
65 *******************************************************************************
66 */
67#define GPDMA_ADDR_CONVERT(addr) \
68 ((((addr) < (OM_MEM_RAM_BASE + CAP_RAM_SIZE)) && ((addr) >= OM_MEM_RAM_BASE)) ? (0x20000000U + (addr & (CAP_RAM_SIZE - 1U))) : addr)

122/// DMA chain transfer, preload the next descriptor before the interrupt is generated
123typedef struct {

138} gpdma_chain_trans_t;
