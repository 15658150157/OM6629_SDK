47/// PMU analog power type
48typedef enum {
49 /// PMU analog power type: RF for calib/test/carrier...
50 PMU_ANA_RF = (1U << 0),
51 /// PMU analog power type: RF only for 24G MAC
52 PMU_ANA_RF_24G = (1U << 1),
53 /// PMU analog power type: RF only for BLE MAC
54 PMU_ANA_RF_BLE = (1U << 2),
55 /// PMU analog power type: GPADC
56 PMU_ANA_GPADC = (1U << 3),
57 /// PMU analog power type: Calibration RC32K
58 PMU_ANA_CALIB_RC32K = (1U << 4),

71/// Mode
72typedef enum {
73 /// Float (Input)
74 PMU_PIN_MODE_FLOAT,
75 /// Push pull (Output)
76 PMU_PIN_MODE_PP,
77 /// Pull up, @ref drv_pmu_pin_pullup_set()
78 PMU_PIN_MODE_PU,

82 PMU_PIN_MODE_OD,

87/// Interrupt trigger type
88typedef enum {

90 PMU_PIN_WAKEUP_LOW_LEVEL,
91 /// Pin wakeup trigger by high level
92 PMU_PIN_WAKEUP_HIGH_LEVEL,

94 /// NOTE: It cannot be used in deep sleep mode with 32k off (@ref drv_pmu_32k_enable_in_deep_sleep)

97 /// NOTE: It cannot be used in deep sleep mode with 32k off (@ref drv_pmu_32k_enable_in_deep_sleep)

99 /// Pin wakeup trigger type is both edge
100 PMU_PIN_WAKEUP_RISING_FAILING_EDGE,

103} pmu_pin_wakeup_type_t;

334/**
335 * @brief Change xtal 32k params
336 *
337 * @param[in] load_capacitance load capacitance, range:0~3, default:0(0pF), step:1.5pF, max:4.5pF
338 * @param[in] drive_current drive current, range:0-3, default:0
339 *
340 * @note load_capacitance will effect xtal 32k startup time and precision,
341 * appropriate value can speed up startup time.
342 **/
343__STATIC_INLINE void drv_pmu_xtal32k_change_param(int load_capacitance, int drive_current)
344{
345 register_set(&OM_PMU->XTAL32K_CTRL, MASK_1REG(PMU_XTAL32K_CTRL_XTAL32K_CTUNE, load_capacitance));
346 register_set(&OM_PMU->CLK_CTRL_2, MASK_1REG(PMU_CLK_CTRL_2_XTAL32K_ISEL, drive_current));
347}
348
349/**
350 *******************************************************************************
351 * @brief Change xtal 32m params
352 *
353 * @param[in] load_capacitance load capacitance, range:0~63, default:32, step:0.38pF(<20ppm), max:23.94pF
354 *
355 * @note load_capacitance will effect xtal 32m precision and frequency offset.
356 *******************************************************************************
357 **/
358__STATIC_INLINE void drv_pmu_xtal32m_change_param(uint32_t load_capacitance)
359{
360 register_set(&OM_PMU->CLK_CTRL_2, PMU_CLK_CTRL_2_CT_XTAL32M_MASK, (load_capacitance << PMU_CLK_CTRL_2_CT_XTAL32M_POS)); // TX
361 register_set(&OM_PMU->OM26B_CFG0, PMU_OM26B_CFG0_XTAL32M_CT_RX_MASK,(load_capacitance << PMU_OM26B_CFG0_XTAL32M_CT_RX_POS)); // RX
362}

371 return (OM_PMU->CLK_CTRL_2 & PMU_CLK_CTRL_2_CT_XTAL32M_MASK) >> PMU_CLK_CTRL_2_CT_XTAL32M_POS;
372}

742/**
743 *******************************************************************************
744 * @brief iflash set power mode
745 *******************************************************************************
746 */
747__STATIC_FORCEINLINE void drv_pmu_iflash_power_mode_set(pmu_iflash_power_mode_t mode)
748{
749 OM_CRITICAL_BEGIN();
750 switch (mode) {
751 case PMU_IFLASH_MANU_POWER_UP:
752 REGW(&OM_PMU->ANA_PD, MASK_2REG(PMU_ANA_PD_PD_FLASH_ME, 1, PMU_ANA_PD_PD_FLASH_REG, 0));
753 while(!drv_pmu_iflash_ready_state_get());
754 break;
755 case PMU_IFLASH_MANU_POWER_DOWN:
756 REGW(&OM_PMU->ANA_PD, MASK_2REG(PMU_ANA_PD_PD_FLASH_ME, 1, PMU_ANA_PD_PD_FLASH_REG, 1));
757 break;
758 case PMU_IFLASH_AUTO_SLEEP_POWER_DOWN_ENABLE:
759 REGW(&OM_PMU->ANA_PD, MASK_2REG(PMU_ANA_PD_PD_FLASH_ME, 0, PMU_ANA_PD_PD_FLASH_REG, 1));
760 break;
761 case PMU_IFLASH_AUTO_SLEEP_POWER_DOWN_DISABLE:
762 REGW(&OM_PMU->ANA_PD, MASK_2REG(PMU_ANA_PD_PD_FLASH_ME, 0, PMU_ANA_PD_PD_FLASH_REG, 0));
763 break;
764 default:
765 break;
766 }
767 OM_CRITICAL_END();
768}
