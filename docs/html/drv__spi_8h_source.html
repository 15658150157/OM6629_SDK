43/// SPI Role
44typedef enum {
45 /// Slave
46 SPI_ROLE_SLAVE = 0,
47 /// Master
48 SPI_ROLE_MASTER = 1,
49} spi_role_t;
50
51/// Valid level of CS
52typedef enum {
53 /// Valid level of CS is Low
54 SPI_CS_LOW = 0,
55 /// Valid level of CS is High
56 SPI_CS_HIGH = 1,
57} spi_cs_t;
58
59/// SPI Mode
60typedef enum {
61 /// MODE0: the idle time is low and the sampling starts from the first hop edge
62 SPI_MODE_0 = 0,
63 /// MODE1: the idle time is low and the sampling starts from the second hop edge
64 SPI_MODE_1 = 1,
65 /// MODE2: the idle time is high and the sampling starts from the first hop edge
66 SPI_MODE_2 = 2,
67 /// MODE3: the idle time is high and the sampling starts from the second hop edge
68 SPI_MODE_3 = 3,
69
70 SPI_MODE_NUM,
71} spi_mode_t;
72
73/// SPI Wire
74typedef enum {
75 /// 3 wire: CS<->CS, CLK<->CLK, DO<->DO
76 SPI_WIRE_3 = 0,
77 /// 4 wire: CS<->CS, CLK<->CLK, DO<->DI, DI<->DO
78 SPI_WIRE_4 = 1,
79} spi_wire_t;
80
81/// SPI MSB LSB Transmission
82typedef enum {
83 /// data transfers start from LSB bit
84 SPI_LSB_FIRST = 0,
85 /// data transfers start from MSB bit
86 SPI_MSB_FIRST = 1,
87} spi_first_t;
88
89/// SPI Control
90typedef enum {
91 SPI_CONTROL_CSN = 0U, /**< Set SPI controller CSn low/high, argu indicate csn value **/
92 SPI_CONTROL_DLY_SAMPLE_FE_SET = 1U, /**< When argu is NULL, disable falling edge delay sample and vice versa **/
93 SPI_CONTROL_DLY_SAMPLE_CYCLE_NUM_SET = 2U, /**< Set delay sample cycle num, argu is in the range [0, 3] **/
94 SPI_CONTROL_TX_INT_MASK_SET = 3U, /**< Set SPI tx total interrupt mask，1 is enable interrupt，0 is mask interrupt **/
95 SPI_CONTROL_RX_INT_MASK_SET = 4U, /**< Set SPI rx total interrupt mask，1 is enable interrupt，0 is mask interrupt **/
96} spi_control_t;
97

100 /// Specifies SPI clock frequency, NOTE: spi_clock must <= cpu_clock / 2

102 /// Specifies SPI Mode

104 /// Specifies SPI Role
105 spi_role_t role;
106 /// Specifies SPI Wire

108 /// Specifies whether data transfers start from LSB or MSB bit
109 spi_first_t first_bit;
110 /// Specifies CS valid level
111 spi_cs_t cs_valid;
112} spi_config_t;

121 *
122 * @param idx Index of SPI peripheral
123 *
124 * @return OM_SPI Type pointer
125 *******************************************************************************
126 */
127static inline OM_SPI_Type* drv_spi_idx2base(uint8_t idx)
128{
129 OM_SPI_Type *const spi[] = {
130 #if (RTE_SPI0)
131 OM_SPI0,
132 #else
133 NULL,
134 #endif
135 #if (RTE_SPI1)
136 OM_SPI1,
137 #else
138 NULL,
139 #endif
140 };
141
142 return (idx < sizeof(spi)/sizeof(spi[0])) ? spi[idx] : NULL;
143}

291/**
292 *******************************************************************************
293 * @brief Control the SPI
294 *
295 * @param[in] om_spi Pointer to SPI
296 * @param[in] control Control options
297 * @param[in] argu argument for control options
298 *
299 *******************************************************************************
300 */
301static inline void drv_spi_control(OM_SPI_Type *om_spi, spi_control_t control, void *argu)
302{
303 switch (control) {
304 case SPI_CONTROL_CSN:
305 om_spi->CSNCTRL = SPI_CSNCTRL_CS_MODE_MASK + ((uint32_t)argu ? SPI_CSNCTRL_CS_GPO_MASK : 0U);
306 break;
307 case SPI_CONTROL_DLY_SAMPLE_FE_SET:
308 register_set(&om_spi->DELAY_CFG, MASK_1REG(SPI_DELAY_CFG_SAMPLE_FE_EN, ((uint32_t)argu) ? 1U : 0U));
309 break;
310 case SPI_CONTROL_DLY_SAMPLE_CYCLE_NUM_SET:
311 OM_ASSERT((uint32_t)argu <= 3);
312 register_set(&om_spi->DELAY_CFG, MASK_1REG(SPI_DELAY_CFG_SAMPLE_CYCLE_NUM, (uint32_t)argu));
313 break;
314 case SPI_CONTROL_TX_INT_MASK_SET:
315 register_set(&om_spi->DELAY_CFG, MASK_1REG(SPI_DELAY_CFG_TX_INT, (uint32_t)argu));
316 break;
317 case SPI_CONTROL_RX_INT_MASK_SET:
318 register_set(&om_spi->DELAY_CFG, MASK_1REG(SPI_DELAY_CFG_RX_INT, (uint32_t)argu));
319 break;
320 default:
321 break;
322 }
323}
